def query_climate_database(db_file, country, year_begin, year_end, month):
    conn = sqlite3.connect(db_file)
    query = f'''
    SELECT S.NAME, S.LATITUDE, S.LONGITUDE, C.NAME, T.YEAR, T.MONTH, T.Temp
    FROM stations S
    LEFT JOIN temperatures T ON T.id = S.id
    LEFT JOIN countries C ON SUBSTRING(T.id, 1, 2) = C.'FIPS 10-4'
    WHERE C.NAME == "{country}" AND T.MONTH == {month} AND YEAR BETWEEN {year_begin} AND {year_end}
    '''

    df = pd.read_sql_query(query, conn)

    conn.close()

    return df

Create Visualization

In this part, we will write a function to create visualizations that address the following question:

How does the average yearly change in temperature vary within a given country?

# we first create a function to pull up relevant information in a datafram
from plotly import express as px
from sklearn.linear_model import LinearRegression
def temperature_coefficient(db_file, country, year_begin, year_end, month, min_obs):
    from climate_database import query_climate_database
    df = query_climate_database(db_file,
                                country, 
                               year_begin, 
                               year_end,
                               month)

    value_counts = (df["NAME"].value_counts() >= min_obs) 
    new_value_counts = value_counts[value_counts==True]
    new_df = df[df['NAME'].isin(new_value_counts.index)].copy()
    new_df["change"] = new_df["NAME"]
    
    for i in new_df['NAME'].unique():
        df2 =  new_df[new_df["NAME"]==i].copy()
        X = df2[["Year"]].copy()
        y = df2['Temp'].copy()
        model = LinearRegression()
        model.fit(X, y)

        # Get the first coefficient
        first_coefficient = model.coef_[0]
        
        new_df.loc[new_df["NAME"]==i, "change"] = first_coefficient
        
    return new_df

# plot the scatter_mapbox
def temperature_coefficient_plot(db_file, country, year_begin, year_end, month, min_obs, **kwargs):
    df = temperature_coefficient(db_file, country, year_begin, year_end, month, min_obs)
    
    df["Estimated Yearly Increase"] = pd.to_numeric(df["change"]).round(4)
    fig = px.scatter_mapbox(df, 
                            lat = "LATITUDE",
                            lon = "LONGITUDE", 
                            hover_name = "NAME", 
                            color = "Estimated Yearly Increase",
                            color_continuous_midpoint = 0,
                            title=f"Estimate of Yearly Change in Temperature of Stations in {country} in month {month} from {year_begin} to {year_end}",
                            **kwargs)
                            
    return fig

def plot_temperature_trend(db_file, country, year_begin, year_end):
    from climate_database import query_average_tempchange
    df = query_average_tempchange(db_file, country, year_begin, year_end)
    fig = px.line(df, x='Year', y='AverageTemp', title=f'Average Yearly Temperature Trend in {country} ({year_begin}-{year_end})')
    fig.update_layout(xaxis_title='Year', yaxis_title='Average Temperature')
    return fig
